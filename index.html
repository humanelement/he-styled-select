Human Element (HE) "Styled Select" Demo

A lightweight plugin to swap out select dropdowns with html that can be styled and hooked up to an onchange handler.

Designate which select dropdowns have this functionality with heStyledSelect.init( ... );

Code for these examples

<script type="text/javascript">
jQuery(document).ready(function(){

    /* ===============================================================
    * Start of simulating a change event outside of the he_styled_select plugin
    * that is already being attached to the <select> element
    * ================================================================ */
    jQuery('select.example1, select.example2').change(function(){ //event attached to the original <select> element
     var selectedVal=jQuery(this).val();
     var selectedOption;
     if(selectedVal==undefined){
        selectedOption=jQuery(this).children('option').not('[value]').eq(0);
        if(selectedOption.length<1){
         selectedOption=jQuery(this).children('option[value=""]:first');
        }
     }else{
        selectedOption=jQuery(this).children('option[value="'+selectedVal+'"]:first');
     }
     //print a message on selection change
     console.log('You selected: "'+selectedOption.text()+'"');
    });
    /* ===============================================================
    * End of simulating a change event outside of the he_styled_select plugin
    * that is already being attached to the <select> element
    * ================================================================ */





    //example 1
    heStyledSelect.init({
        select:'select.example1' //select a specific <select> element to "style"
        //no onchange is specified, so NO event is triggered onchange
    });

    //example 2
    heStyledSelect.init({
        select:'select.example2', //select a specific <select> element to "style"
        onchange:function(select, args){
            select.change(); //trigger the original <select> jQuery change event
            //Event.simulate(select[0], 'change'); //prototype
        }
    });
});
</script>
